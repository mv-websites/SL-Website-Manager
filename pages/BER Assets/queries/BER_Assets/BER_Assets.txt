SELECT Equip_Status_Description as serviceable, 
  Customer,
	Client_Group,
	Asset_Number as asset_number, 
	serial_number, 
	CONCAT(Make,  ' - ', Model) as make_model, 
	description, 
	Location as location, 
	CONCAT('[', Post_Code, ']', ' - ', Co_Name) as site, 
	Post_Code as postcode
FROM tempdev_.LiveEquipment_SL WHERE Equip_Status_Description = "BER";
